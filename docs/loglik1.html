edge1 <- rbind(c(0,0),c(1,0))
edge2 <- rbind(c(0,0),c(0,1))
edge3 <- rbind(c(0,1),c(-1,1))
theta <- seq(from=pi,to=3*pi/2,length.out = 20)
edge4 <- cbind(sin(theta),1+ cos(theta))
edges = list(edge1, edge2, edge3, edge4)
graph <- metric_graph$new(edges = edges)
graph$build_mesh(h = 0.01)
sigma <- 1.3
range <- 0.15
nu <- 0.8 
rspde.order <- 2
op <- matern.operators(nu = nu, range = range, sigma = sigma, 
                       parameterization = "matern",
                       m = rspde.order, graph = graph)  
u <- simulate(op)

obs.per.edge <- 100
obs.loc <- NULL
for(i in 1:graph$nE) {
  obs.loc <- rbind(obs.loc,
                   cbind(rep(i,obs.per.edge), runif(obs.per.edge)))
}
n.obs <- obs.per.edge*graph$nE
A <- graph$fem_basis(obs.loc)
sigma.e <- 0.1
Y <- as.vector(A %*% u + sigma.e * rnorm(n.obs))
df_data <- data.frame(y = Y, edge_number = obs.loc[,1],
                        distance_on_edge = obs.loc[,2])
graph$add_observations(data = df_data, normalized = TRUE)
fit <- graph_lme(y ~ -1, graph=graph, BC = 0,
            model = list(type = "WhittleMatern", alpha = 1))

summary(fit)

## [1] -189.4241
sigma_e <- fit$coeff$measurement_error
alpha <- 1
nu <- alpha - 1/2
tau <- fit$coeff$random_effects[1]
kappa <- fit$coeff$random_effects[2]

m <- graph$nV
n <- length(Y)


graph$observation_to_vertex()

data <- graph$get_data()

Y <- data$y

Q <- MetricGraph:::Qalpha1(
      theta = c(tau, kappa),
      graph = graph,
      BC = 0,
      build = TRUE)


barA <- cbind(matrix(0, nrow = n, ncol = m),
              Matrix::Diagonal(n))

Qp <- Q + 1/sigma_e^2 * t(barA) %*% barA 
mu <- 1/sigma_e^2 * solve(Qp, t(barA) %*% Y)

L <- chol(Q)
Lp <- chol(Qp)

loglik <- sum(log(diag(L))) +
  n*log(sigma_e) -
  sum(log(diag(Lp))) -
  0.5/sigma_e^2 * t(Y) %*% Y +
  0.5 * t(mu) %*% Qp %*% mu

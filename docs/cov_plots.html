# Function to build an interval graph and create a mesh
gets_graph_interval <- function(h, ell){
  edge <- rbind(c(0,0),c(ell,0))
  edges = list(edge)
  graph <- metric_graph$new(edges = edges)
  graph$build_mesh(h = h)
  return(graph)
}
# Parameters of the approximation
h = 0.02
ell = 2

graph$compute_fem()

nodes <- graph$mesh$V[,1]

r_bridge <- function(t1, t2, kappa, alpha, sigma, ell){
  nu <- alpha - 1/2
  r1_t1_0 <- sigma^2*matern.p.joint(s = t1, t = 0, kappa = kappa, p = 0, alpha = alpha)[1,]
  r1_t1_ell <- sigma^2*matern.p.joint(s = t1, t = ell, kappa = kappa, p = 0, alpha = alpha)[1,]
  
  r_0_0 <- sigma^2*matern.p.joint(s = 0, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r_0_ell <- sigma^2*matern.p.joint(s = 0, t = ell, kappa = kappa, p = 0, alpha = alpha)
  r_ell_0 <- sigma^2*matern.p.joint(s = ell, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r_ell_ell <- sigma^2*matern.p.joint(s = ell, t = ell, kappa = kappa, p = 0, alpha = alpha)
  
  r1_0_t2 <- sigma^2*matern.p.joint(s = 0, t = t2, kappa = kappa, p = 0, alpha = alpha)[1,]
  r1_ell_t2 <- sigma^2*matern.p.joint(s = ell, t = t2, kappa = kappa, p = 0, alpha = alpha)[1,]
  
  r1_h <- matrix(c(r1_t1_0, r1_t1_ell), nrow = 1, ncol = 2*alpha)
  r1_v <- matrix(c(r1_0_t2, r1_ell_t2), nrow = 2*alpha, ncol = 1)
  
  cov <- matern.covariance(h = t1 - t2, kappa = kappa, nu = nu, sigma = sigma) - r1_h %*% solve(rbind(cbind(r_0_0, r_0_ell), cbind(r_ell_0, r_ell_ell)), r1_v)
  return(as.numeric(cov))
}
r_hat <- function(t1, t2, kappa, alpha, sigma, ell){
  nu <- alpha - 1/2
  r1_t1_0 <- sigma^2*matern.p.joint(s = t1, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r1_t1_ell <- sigma^2*matern.p.joint(s = t1, t = ell, kappa = kappa, p = 0, alpha = alpha)
  
  r_0_0 <- sigma^2*matern.p.joint(s = 0, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r_0_ell <- -sigma^2*matern.p.joint(s = 0, t = ell, kappa = kappa, p = 0, alpha = alpha)
  r_ell_0 <- -sigma^2*matern.p.joint(s = ell, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r_ell_ell <- sigma^2*matern.p.joint(s = 0, t = 0, kappa = kappa, p = 0, alpha = alpha)
  
  r1_0_t2 <- sigma^2*matern.p.joint(s = t2, t = 0, kappa = kappa, p = 0, alpha = alpha)
  r1_ell_t2 <- sigma^2*matern.p.joint(s = t2, t = ell, kappa = kappa, p = 0, alpha = alpha)
  
  
  cov <- sigma^2*matern.p.joint(s = t1, t = t2, kappa = kappa, p = 0, alpha = alpha) + cbind(r1_t1_0, r1_t1_ell) %*% solve(rbind(cbind(r_0_0, r_0_ell), cbind(r_ell_0, r_ell_ell)), rbind(r1_0_t2, r1_ell_t2))
  return(as.numeric(cov[1,1]))
}
